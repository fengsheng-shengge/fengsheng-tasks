## 三、C2 · 评分算法 + 档位判定逻辑

### 3.1 5题评分规则

| 选项 | 分值 | 含义 |
|------|------|------|
| A | 3分 | 最佳实践（L2水平） |
| B | 2分 | 合格做法（L1水平） |
| C | 1分 | 常见误区（L0水平） |
| D | 0分 | 严重错误（零容忍） |

### 3.2 档位判定规则

| 总分范围 | 档位 | 特征 | 推荐学习路径 |
|---------|------|------|------------|
| 0-5分 | **L0 本能型** | 凭感觉做事，没有方法论 | 全部3课：第1课→第2课→第3课 |
| 6-9分 | **L1 标准型** | 能按流程做，但不灵活 | 2-3课：从第2课开始→第3课 |
| 10-15分 | **L2 专业型** | 能独立判断，会应变 | 第3课+直接进练模块 |

### 3.3 薄弱维度标记规则

- 每道题对应1个诊断维度（D1/D2/D3/D4）
- 该维度只有1道题，得分即为该维度分（0-3分）
- **标记规则**：该维度得分 ≤ 1分 → 标记"需强化"
- 维度映射表：

| 题目ID | 诊断维度 |
|--------|---------|
| Q1_COMMON_D1 | D1客户识别 |
| Q2_COMMON_D3 | D3信任建立 |
| Q3_TRADE_D1 / Q3_RENTAL_D1 | D1客户识别 |
| Q4_TRADE_D2 / Q4_RENTAL_D2 | D2需求挖掘 |
| Q5_TRADE_D4 / Q5_RENTAL_D4 | D4场景应对 |

**注意**：通用题Q1和分化题Q3均对应D1，D1维度取两题平均分（取整）。

### 3.4 推荐学习路径映射

| 档位 | 薄弱维度 | 推荐路径 |
|------|---------|---------|
| L0 | — | 第1课→第2课→第3课（全修） |
| L0 | D1/D2 | 第1课→第2课→第3课，重点练D1/D2场景 |
| L0 | D3/D4 | 第1课→第2课→第3课，重点练D3/D4场景 |
| L1 | D1/D2需强化 | 从第2课开始→第3课，补第1课核心概念 |
| L1 | D3/D4需强化 | 从第2课开始→第3课，重点学第3课信任与闭环 |
| L1 | 无薄弱 | 第2课→第3课，巩固后进练 |
| L2 | — | 直接进练模块，第3课选学 |
| L2 | D4需强化 | 第3课必学（场景应对），然后进练 |

### 3.5 评分函数（JS伪代码）

```javascript
/**
 * 客户解码 · 入学诊断评分函数
 * 输入：5道题的答案 + 身份线
 * 输出：档位 + 各维度得分 + 薄弱维度 + 推荐学习路径
 */

// ========== 题目配置 ==========
const QUESTIONS = {
  // 通用题（双线共用）
  Q1: { id: 'Q1_COMMON_D1', dimension: 'D1', type: 'common' },
  Q2: { id: 'Q2_COMMON_D3', dimension: 'D3', type: 'common' },
  // 交易线分化题
  Q3_TRADE: { id: 'Q3_TRADE_D1', dimension: 'D1', type: 'trade' },
  Q4_TRADE: { id: 'Q4_TRADE_D2', dimension: 'D2', type: 'trade' },
  Q5_TRADE: { id: 'Q5_TRADE_D4', dimension: 'D4', type: 'trade' },
  // 租赁线分化题
  Q3_RENTAL: { id: 'Q3_RENTAL_D1', dimension: 'D1', type: 'rental' },
  Q4_RENTAL: { id: 'Q4_RENTAL_D2', dimension: 'D2', type: 'rental' },
  Q5_RENTAL: { id: 'Q5_RENTAL_D4', dimension: 'D4', type: 'rental' },
};

// 选项→分值映射
const SCORE_MAP = { A: 3, B: 2, C: 1, D: 0 };

// 档位阈值
const GRADE_THRESHOLDS = [
  { min: 10, max: 15, grade: 'L2', label: '专业型' },
  { min: 6,  max: 9,  grade: 'L1', label: '标准型' },
  { min: 0,  max: 5,  grade: 'L0', label: '本能型' },
];

// ========== 主评分函数 ==========
function evaluateDiagnostic(answers, identity) {
  // identity: 'trade' | 'rental'
  // answers: { Q1: 'A', Q2: 'B', Q3: 'A', Q4: 'C', Q5: 'B' }

  // Step 1: 计算总分
  const answerKeys = Object.keys(answers);
  let totalScore = 0;
  answerKeys.forEach(key => {
    totalScore += SCORE_MAP[answers[key]] || 0;
  });

  // Step 2: 计算各维度得分
  const dimensionScores = { D1: [], D2: [], D3: [], D4: [] };

  // 通用题
  dimensionScores.D1.push(SCORE_MAP[answers.Q1]);
  dimensionScores.D3.push(SCORE_MAP[answers.Q2]);

  // 分化题
  if (identity === 'trade') {
    dimensionScores.D1.push(SCORE_MAP[answers.Q3]);
    dimensionScores.D2.push(SCORE_MAP[answers.Q4]);
    dimensionScores.D4.push(SCORE_MAP[answers.Q5]);
  } else {
    dimensionScores.D1.push(SCORE_MAP[answers.Q3]);
    dimensionScores.D2.push(SCORE_MAP[answers.Q4]);
    dimensionScores.D4.push(SCORE_MAP[answers.Q5]);
  }

  // 维度平均分（D1有两题，取平均取整；其他维度单题直接取）
  const dimensions = {};
  Object.keys(dimensionScores).forEach(dim => {
    const scores = dimensionScores[dim];
    if (scores.length > 0) {
      dimensions[dim] = Math.round(
        scores.reduce((sum, s) => sum + s, 0) / scores.length
      );
    } else {
      dimensions[dim] = 0; // 该线未覆盖的维度
    }
  });

  // Step 3: 判定档位
  let grade = 'L0';
  let gradeLabel = '本能型';
  for (const threshold of GRADE_THRESHOLDS) {
    if (totalScore >= threshold.min && totalScore <= threshold.max) {
      grade = threshold.grade;
      gradeLabel = threshold.label;
      break;
    }
  }

  // Step 4: 标记薄弱维度（得分 ≤ 1分）
  const weakDimensions = [];
  Object.keys(dimensions).forEach(dim => {
    if (dimensions[dim] <= 1) {
      weakDimensions.push({
        dimension: dim,
        score: dimensions[dim],
        label: getDimensionLabel(dim),
        status: '需强化'
      });
    }
  });

  // Step 5: 推荐学习路径
  const learningPath = getLearningPath(grade, weakDimensions);

  // Step 6: 组装结果
  const result = {
    totalScore,
    maxScore: 15,
    grade,
    gradeLabel,
    dimensions,
    weakDimensions,
    learningPath,
    identity,
    evaluatedAt: new Date().toISOString()
  };

  // Step 7: 存入 localStorage
  saveToLocalStorage(result);

  return result;
}

// ========== 辅助函数 ==========

function getDimensionLabel(dim) {
  const labels = {
    D1: '客户识别',
    D2: '需求挖掘',
    D3: '信任建立',
    D4: '场景应对'
  };
  return labels[dim] || dim;
}

function getLearningPath(grade, weakDimensions) {
  const weakDims = weakDimensions.map(w => w.dimension);

  if (grade === 'L0') {
    return {
      description: '从方法论底子开始——你需要建立完整的客户解码体系',
      lessons: ['lesson1', 'lesson2', 'lesson3'],
      focus: weakDims.length > 0
        ? `重点强化：${weakDims.map(d => getDimensionLabel(d)).join('、')}`
        : '全部课程必修',
      practiceEntry: '学完全部3课后进入练模块'
    };
  }

  if (grade === 'L1') {
    if (weakDims.includes('D1') || weakDims.includes('D2')) {
      return {
        description: '补上方法论的短板——你有一定基础，但客户识别和需求挖掘还需加强',
        lessons: ['lesson2', 'lesson3'],
        supplement: 'lesson1',
        focus: '重点强化：客户识别、需求挖掘',
        practiceEntry: '学完第2-3课后进入练模块'
      };
    }
    if (weakDims.includes('D3') || weakDims.includes('D4')) {
      return {
        description: '信任和场景应对是短板——学会把解码变成客户愿意买单的方案',
        lessons: ['lesson2', 'lesson3'],
        focus: '重点强化：信任建立、场景应对',
        practiceEntry: '第3课必学，学完进练模块'
      };
    }
    return {
      description: '你有方法论但还不够灵活——补上第2-3课，巩固后实战',
      lessons: ['lesson2', 'lesson3'],
      focus: '巩固方法论+工具使用',
      practiceEntry: '学完第2-3课后进入练模块'
    };
  }

  // L2
  if (weakDims.includes('D4')) {
    return {
      description: '整体不错，但场景应对还需打磨——直接进练，第3课选学',
      lessons: ['lesson3'],
      optional: true,
      focus: '场景应对实战强化',
      practiceEntry: '直接进入练模块，第3课选学'
    };
  }
  return {
    description: '你有方法论，实战巩固——直接进练模块',
    lessons: ['lesson3'],
    optional: true,
    focus: '实战巩固',
    practiceEntry: '直接进入练模块，第3课选学'
  };
}

function saveToLocalStorage(result) {
  const storageKey = 'customerDecoding_diagnostic';
  const historyKey = 'customerDecoding_gradeHistory';

  // 存最新诊断结果
  localStorage.setItem(storageKey, JSON.stringify(result));

  // 追加档位历史
  const history = JSON.parse(localStorage.getItem(historyKey) || '[]');
  history.push({
    grade: result.grade,
    date: result.evaluatedAt,
    source: 'test',
    totalScore: result.totalScore
  });
  localStorage.setItem(historyKey, JSON.stringify(history));

  // 更新用户档案
  const profileKey = 'customerDecoding_userProfile';
  const profile = JSON.parse(localStorage.getItem(profileKey) || '{}');
  profile.grade = { current: result.grade, lastUpdated: result.evaluatedAt };
  profile.dimensions = result.dimensions;
  localStorage.setItem(profileKey, JSON.stringify(profile));
}

// ========== 调用示例 ==========
// 交易线示例
const tradeResult = evaluateDiagnostic(
  { Q1: 'A', Q2: 'B', Q3: 'B', Q4: 'A', Q5: 'C },
  'trade'
);
// 预期输出: totalScore=9, grade='L1', dimensions={D1:2, D2:3, D3:2, D4:1}, weakDimensions=[{D4,1,'需强化'}]

// 租赁线示例
const rentalResult = evaluateDiagnostic(
  { Q1: 'A', Q2: 'A', Q3: 'A', Q4: 'C', Q5: 'A' },
  'rental'
);
// 预期输出: totalScore=11, grade='L2', dimensions={D1:3, D2:1, D3:3, D4:3}, weakDimensions=[{D2,1,'需强化'}]
```

## 四、前端出题逻辑说明

```
用户进入测评页
    → 选择身份（经纪人/客户经理）
    → localStorage存身份
    → 渲染5道题：
        Q1（通用D1）+ Q2（通用D3）+ Q3~Q5（按身份从题库切换）
    → 用户提交答案
    → 调用 evaluateDiagnostic(answers, identity)
    → 展示结果页：
        - 档位标签（L0/L1/L2）+ 雷达图（D1-D4）
        - 薄弱维度（标红"需强化"）
        - 推荐学习路径（点击进入对应微课）
```
