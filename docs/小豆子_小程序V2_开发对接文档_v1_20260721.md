### 3.3 搜索别名对接

```js
// 词典搜索逻辑
function searchEntries(query, entries, aliases) {
  const lowerQuery = query.toLowerCase()
  // 1. 直接匹配词条名
  const nameMatches = entries.filter(e => e.name.toLowerCase().includes(lowerQuery))
  // 2. 通过别名匹配
  const aliasMatches = aliases
    .filter(a => a.alias.toLowerCase().includes(lowerQuery))
    .map(a => entries.find(e => e.id === a.entryId))
    .filter(Boolean)
  // 去重合并
  return [...new Set([...nameMatches, ...aliasMatches])]
}
```

## 六、5 轮免费 → 付费弹窗逻辑

### 6.1 触发条件

```js
// 导师页每次发送消息前检查
function checkQuotaBeforeSendMessage() {
  const userData = wx.getStorageSync('fs_user_data')
  const today = new Date().toISOString().slice(0, 10)

  // 跨日重置
  if (userData.lastChatResetDate !== today) {
    userData.freeChatCount = 5
    userData.lastChatResetDate = today
    wx.setStorageSync('fs_user_data', userData)
  }

  // 已订阅用户无限次
  if (userData.subscription && userData.subscription.expireAt > Date.now()) {
    return { canSend: true }
  }

  // 免费额度内
  if (userData.freeChatCount > 0) {
    userData.freeChatCount--
    wx.setStorageSync('fs_user_data', userData)
    return { canSend: true, remaining: userData.freeChatCount }
  }

  // 额度用完，弹付费墙
  return { canSend: false, reason: 'free_quota_exhausted' }
}
```

## 九、设计决策回溯表

| 设计决策 | 回溯到设计共识铁律 |
|---|---|
| 4 张场景卡片而非 4 张功能卡片 | 定位翻转：场景洞察+转化工具双驱动 |
| 词条详情页专业理解+法律依据免费 | 积分底线：基础内容永远免费 |
| 导师对话必须附带依据 | 言之有物：每个输出必须有依据 |
| 游戏化每 3-5 关穿插知识 | 趣味性是手段，专业内容是目的 |
| 积分获取只奖励专业成长行为 | 用户贡献飞轮：时间不可压缩 |
| 场景卡片含客户原话 | 理解先于应对：先理解客户真正在顾虑什么 |
| 贡献署名展示 | 用户贡献飞轮：被收录+署名→其他人看到→也贡献 |
| 见客户时只查政策/数据 | 在客户面前掏手机=只有查政策/数据给客户看是正当场景 |
| 5 轮免费额度 | 提审合规：不强制付费，免费路径必须够用 |
| 免费月卡 2000 积分兑换 | 用户贡献飞轮：活跃用户可免费使用 |
